\Notion{
    Summary
}{
    Sets define ordered and unordered pairs.
    Ordered pairs contain coordinates.
    Coordinates define equivalent.
}

\Definition{
    Unordered Pair
}{
    \Set{$a, b$} is a set whose elements are exactly $a$ and $b$.
}
\Definition{
    Ordered Pair
}{
    Ordered pair of $a$ and $b$ is denoted by $(a, b)$.
}
\Definition{
    Coordinate
}{
    $a$ is the first coordinate of the pair, $b$ is the second coordinate.
}
\Definition{}{
    $(a, b)$ $=$ \Set{\Set{$a$}, \Set{$a, b$}}.
}
\Definition{}{
    For two different sets $\alpha$ and $\beta$, $(a, b)$ $=$ \Set{\Set{$a, \alpha$}, \Set{$b, \beta$}}.
}
\Note{
    An alternative definition of ordered pairs.
}

\Theorem{}{
    \Keywords{$(a, b)$ $=$ $(a', b')$} if and only if \Keywords{$a$ $=$ $a'$} and \Keywords{$b$ $=$ $b'$}.
}{
    Obviously.
}
\Proposition{}{
    Two \Keywords{ordered pairs} are \Keywords{equal} if and only if their \Keywords{first coordinates} are equal and their \Keywords{second coordinates} are equal.
}{
    Obviously.
}
\Note{
    Ordered pair should be defined in such a way that two ordered pairs are equal if and only if their first coordinates are equal and their second coordinates are equal.
}
\Proposition{}{
    \Keywords{$(a, b)$ $\neq$ $(b, a)$} if \Keywords{$a$ $\neq$ $b$}.
}{
    Obviously.
}
\Proposition{}{
    If \Keywords{$a$ $\neq$ $b$}, $(a, b)$ has two elements, a \Keywords{singleton} \Set{$a$} and an \Keywords{unordered pair} \Set{$a, b$}.
    We find the \Keywords{first coordinate} by looking at the element of \Set{$a$}.
    The \Keywords{second coordinate} is then the other element of \Set{$a, b$}.
    If \Keywords{$a$ $=$ $b$}, then \Keywords{$(a, a)$ $=$ \Set{\Set{$a$}}} has only one element.
}{
    Obviously.
}
\Note{
    If $a \neq b$, $(a, b)$ has two elements, a singleton \Set{$a$} and an unordered pair \Set{$a, b$}.
    The first coordinate is the element of \Set{$a$}; the second coordinate is the other element of \Set{$a, b$}.
}

\Definition{
    One-Tuples
}{
    $(a)$ $=$ $a$.
}
\Definition{
    Ordered Triples
}{
    $(a, b, c)$ $=$ $((a, b), c)$.
}
\Definition{
    Ordered Quadruples
}{
    $(a, b, c, d)$ $=$ $(((a, b), c), d)$.
}

\Property{}{
    $(a, b)$ $\in$ $\Powerset{\Powerset{\Set{a, b}}}$.\
    More generally, if $a$ $\in$ $A$ and $b$ $\in$ $A$, then $(a, b)$ $\in$ $\Powerset{\Powerset{\Set{a, b}}}$.\
}{
    Obviously.
}
\Property{}{
    $a$, $b$ $\in$ $\bigcup(a, b)$.
}{
    Obviously.
}
\Property{}{
    $(a, b)$, $(a, b, c)$, and $(a, b, c, d)$ exist for all $a$, $b$, $c$, and $d$.
}{
    Obviously.
}
\Property{}{
    If $(a, b)$ $=$ $(b, a)$, then $a$ $=$ $b$.
}{
    Obviously.
}
\Property{}{
    $(a, b, c)$ $=$ $(a', b' ,c')$ implies $a$ $=$ $a'$, $b$ $=$ $b'$, and $c$ $=$ $c'$.
    $(a, b, c, d)$ $=$ $(a', b' ,c', d')$ implies $a$ $=$ $a'$, $b$ $=$ $b'$, $c$ $=$ $c'$, and $d$ $=$ $d'$.
}{
    Obviously.
}
